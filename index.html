Connect Supabase

Sync your library across devices by connecting a free Supabase project. Your data is stored in your own database — OnLIB never sees it.

🔑 Sync Code

Your sync code is your shared identity across devices. Enter the same code on every device to share one library. Anyone who knows your code and your Supabase credentials can access your data — keep it private.

1. Run this SQL in your Supabase SQL Editor:

create table if not exists onlib_books (
  id          text primary key,
  user_id     text not null,
  title       text,
  author      text,
  cover       text,
  description text,
  tags        text[],
  pages       integer,
  language    text,
  year        text,
  publisher   text,
  rating      integer default 5,
  folder      text default 'Reading',
  notes       text,
  created_at  timestamptz default now()
);

create table if not exists onlib_profiles (
  user_id    text primary key,
  name       text,
  bio        text,
  pfp        text,
  updated_at timestamptz default now()
);

-- Disable RLS for personal use
alter table onlib_books    disable row level security;
alter table onlib_profiles disable row level security;
